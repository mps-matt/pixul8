611  class PixelOperationWrapper :public ParallelLoopBody
612  {
613  public:
614  PixelOperationWrapper(Mat_<_Tp>* const frame, const Functor& _operation)
615  : mat(frame), op(_operation) {}
616  virtual ~PixelOperationWrapper(){}
617  // ! Overloaded virtual operator
618  // convert range call to row call.
619  virtual void operator()(const Range &range) const CV_OVERRIDE
620  {
621  const int DIMS = mat->dims;
622  const int COLS = mat->size[DIMS - 1];
623  if (DIMS <= 2) {
624  for (int row = range.start; row < range.end; ++row) {
625  this->rowCall2(row, COLS);
626  }
627  } else {
628  std::vector<int> idx(DIMS);
629  idx[DIMS - 2] = range.start - 1;
630 
631  for (int line_num = range.start; line_num < range.end; ++line_num) {
632  idx[DIMS - 2]++;
633  for (int i = DIMS - 2; i >= 0; --i) {
634  if (idx[i] >= mat->size[i]) {
635  idx[i - 1] += idx[i] / mat->size[i];
636  idx[i] %= mat->size[i];
637  continue; // carry-over;
638  }
639  else {
640  break;
641  }
642  }
643  this->rowCall(&idx[0], COLS, DIMS);
644  }
645  }
646  }
647  private:
648  Mat_<_Tp>* const mat;
649  const Functor op;
650  // ! Call operator for each elements in this row.
651  inline void rowCall(int* const idx, const int COLS, const int DIMS) const {
652  int &col = idx[DIMS - 1];
653  col = 0;
654  _Tp* pixel = &(mat->template at<_Tp>(idx));
655 
656  while (col < COLS) {
657  op(*pixel, const_cast<const int*>(idx));
658  pixel++; col++;
659  }
660  col = 0;
661  }
662  // ! Call operator for each elements in this row. 2d mat special version.
663  inline void rowCall2(const int row, const int COLS) const {
664  union Index{
665  int body[2];
666  operator const int*() const {
667  return reinterpret_cast<const int*>(this);
668  }
669  int& operator[](const int i) {
670  return body[i];
671  }
672  } idx = {{row, 0}};
673  // Special union is needed to avoid
674  // "error: array subscript is above array bounds [-Werror=array-bounds]"
675  // when call the functor `op` such that access idx[3].
676 
677  _Tp* pixel = &(mat->template at<_Tp>(idx));
678  const _Tp* const pixel_end = pixel + COLS;
679  while(pixel < pixel_end) {
680  op(*pixel++, static_cast<const int*>(idx));
681  idx[1]++;
682  }
683  }
684  PixelOperationWrapper& operator=(const PixelOperationWrapper &) {
685  CV_Assert(false);
686  // We can not remove this implementation because Visual Studio warning C4822.
687  return *this;
688  }
689  };
690 
691  parallel_for_(cv::Range(0, LINES), PixelOperationWrapper(reinterpret_cast<Mat_<_Tp>*>(this), operation));
692 }

1120 // Basic Node class for tree building
1121 template<class OBJECT>
1122 class CV_EXPORTS Node
1123 {
1124 public:
1125  Node()
1126  {
1127  m_pParent = 0;
1128  }
1129  Node(OBJECT& payload) : m_payload(payload)
1130  {
1131  m_pParent = 0;
1132  }
1133  ~Node()
1134  {
1135  removeChilds();
1136  if (m_pParent)
1137  {
1138  int idx = m_pParent->findChild(this);
1139  if (idx >= 0)
1140  m_pParent->m_childs.erase(m_pParent->m_childs.begin() + idx);
1141  }
1142  }
1143 
1144  Node<OBJECT>* findChild(OBJECT& payload) const
1145  {
1146  for(size_t i = 0; i < this->m_childs.size(); i++)
1147  {
1148  if(this->m_childs[i]->m_payload == payload)
1149  return this->m_childs[i];
1150  }
1151  return NULL;
1152  }
1153 
1154  int findChild(Node<OBJECT> *pNode) const
1155  {
1156  for (size_t i = 0; i < this->m_childs.size(); i++)
1157  {
1158  if(this->m_childs[i] == pNode)
1159  return (int)i;
1160  }
1161  return -1;
1162  }
1163 
1164  void addChild(Node<OBJECT> *pNode)
1165  {
1166  if(!pNode)
1167  return;
1168 
1169  CV_Assert(pNode->m_pParent == 0);
1170  pNode->m_pParent = this;
1171  this->m_childs.push_back(pNode);
1172  }
1173 
1174  void removeChilds()
1175  {
1176  for(size_t i = 0; i < m_childs.size(); i++)
1177  {
1178  m_childs[i]->m_pParent = 0; // avoid excessive parent vector trimming
1179  delete m_childs[i];
1180  }
1181  m_childs.clear();
1182  }
1183 
1184  int getDepth()
1185  {
1186  int count = 0;
1187  Node *pParent = m_pParent;
1188  while(pParent) count++, pParent = pParent->m_pParent;
1189  return count;
1190  }
1191 
1192 public:
1193  OBJECT m_payload;
1194  Node<OBJECT>* m_pParent;
1195  std::vector<Node<OBJECT>*> m_childs;
1196 };
