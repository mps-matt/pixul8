452 template<typename _Tp, typename _AccTp> static inline
453 _AccTp normL2Sqr(const _Tp* a, int n)
454 {
455  _AccTp s = 0;
456  int i=0;
457 #if CV_ENABLE_UNROLLED
458  for( ; i <= n - 4; i += 4 )
459  {
460  _AccTp v0 = a[i], v1 = a[i+1], v2 = a[i+2], v3 = a[i+3];
461  s += v0*v0 + v1*v1 + v2*v2 + v3*v3;
462  }
463 #endif
464  for( ; i < n; i++ )
465  {
466  _AccTp v = a[i];
467  s += v*v;
468  }
469  return s;
470 }
