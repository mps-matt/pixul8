104 enum RegionLocationFlag {
105  REGION_FLAG_FUNCTION = (1 << 0), //< region is function (=1) / nested named region (=0)
106  REGION_FLAG_APP_CODE = (1 << 1), //< region is Application code (=1) / OpenCV library code (=0)
107  REGION_FLAG_SKIP_NESTED = (1 << 2), //< avoid processing of nested regions
108 
109  REGION_FLAG_IMPL_IPP = (1 << 16), //< region is part of IPP code path
110  REGION_FLAG_IMPL_OPENCL = (2 << 16), //< region is part of OpenCL code path
111  REGION_FLAG_IMPL_OPENVX = (3 << 16), //< region is part of OpenVX code path
112 
113  REGION_FLAG_IMPL_MASK = (15 << 16),
114 
115  REGION_FLAG_REGION_FORCE = (1 << 30),
116  REGION_FLAG_REGION_NEXT = (1 << 31), //< close previous region (see #CV_TRACE_REGION_NEXT macro)

118  ENUM_REGION_FLAG_FORCE_INT = INT_MAX
